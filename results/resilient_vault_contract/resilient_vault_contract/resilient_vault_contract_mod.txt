// Resilient vault-style standalone Rust contract.
// This file is intentionally branch-heavy while keeping arithmetic saturating and bounded.

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum VaultDecision {
    Accept,
    Delay,
    Reject,
    Escalate,
}

#[derive(Clone, Copy, Debug)]
pub struct VaultSnapshot {
    pub liquidity: i64,
    pub debt: i64,
    pub trust: i32,
    pub locked: bool,
}

pub fn normalize_amount(amount: i64) -> i64 {
    if amount < 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(amount < 0), "REACHABLE_TRUE");
	assert!((amount < 0), "REACHABLE_FALSE");
        return 0;
    }
    if amount > 1_000_000 {
	// KANI INJECTION: Check Reachability
	assert!(!(amount > 1_000_000), "REACHABLE_TRUE");
	assert!((amount > 1_000_000), "REACHABLE_FALSE");
        return 1_000_000;
    }
    amount
}

pub fn evaluate_vault(snapshot: VaultSnapshot, request: i64, age_days: u32) -> VaultDecision {
    let amount = normalize_amount(request);
    let mut flags = 0i32;

    if snapshot.locked {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.locked), "REACHABLE_TRUE");
	assert!((snapshot.locked), "REACHABLE_FALSE");
        flags += 8;
    }
    if snapshot.liquidity < amount {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.liquidity < amount), "REACHABLE_TRUE");
	assert!((snapshot.liquidity < amount), "REACHABLE_FALSE");
        flags += 5;
    }
    if snapshot.debt > snapshot.liquidity {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.debt > snapshot.liquidity), "REACHABLE_TRUE");
	assert!((snapshot.debt > snapshot.liquidity), "REACHABLE_FALSE");
        flags += 4;
    }
    if snapshot.trust < 20 {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.trust < 20), "REACHABLE_TRUE");
	assert!((snapshot.trust < 20), "REACHABLE_FALSE");
        flags += 3;
    }
    if snapshot.trust > 80 {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.trust > 80), "REACHABLE_TRUE");
	assert!((snapshot.trust > 80), "REACHABLE_FALSE");
        flags -= 2;
    }
    if age_days < 3 {
	// KANI INJECTION: Check Reachability
	assert!(!(age_days < 3), "REACHABLE_TRUE");
	assert!((age_days < 3), "REACHABLE_FALSE");
        flags += 2;
    }
    if age_days > 365 {
	// KANI INJECTION: Check Reachability
	assert!(!(age_days > 365), "REACHABLE_TRUE");
	assert!((age_days > 365), "REACHABLE_FALSE");
        flags -= 1;
    }
    if amount == 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(amount == 0), "REACHABLE_TRUE");
	assert!((amount == 0), "REACHABLE_FALSE");
        flags += 1;
    }
    if amount > 100_000 && snapshot.trust < 60 {
	// KANI INJECTION: Check Reachability
	assert!(!(amount > 100_000 && snapshot.trust < 60), "REACHABLE_TRUE");
	assert!((amount > 100_000 && snapshot.trust < 60), "REACHABLE_FALSE");
        flags += 4;
    }
    if snapshot.liquidity.saturating_sub(snapshot.debt) > amount {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.liquidity.saturating_sub(snapshot.debt) > amount), "REACHABLE_TRUE");
	assert!((snapshot.liquidity.saturating_sub(snapshot.debt) > amount), "REACHABLE_FALSE");
        flags -= 2;
    }

    if flags >= 10 {
	// KANI INJECTION: Check Reachability
	assert!(!(flags >= 10), "REACHABLE_TRUE");
	assert!((flags >= 10), "REACHABLE_FALSE");
        VaultDecision::Reject
    } else if flags >= 6 {
        VaultDecision::Escalate
    } else if flags >= 2 {
        VaultDecision::Delay
    } else {
        VaultDecision::Accept
    }
}

pub fn apply_decision(snapshot: VaultSnapshot, decision: VaultDecision, request: i64) -> VaultSnapshot {
    let mut next = snapshot;
    let amount = normalize_amount(request);

    if decision == VaultDecision::Accept {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Accept), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Accept), "REACHABLE_FALSE");
        next.liquidity = next.liquidity.saturating_sub(amount);
        next.debt = next.debt.saturating_add(amount / 4);
        next.trust = next.trust.saturating_add(1);
    }
    if decision == VaultDecision::Delay {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Delay), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Delay), "REACHABLE_FALSE");
        next.trust = next.trust.saturating_sub(1);
    }
    if decision == VaultDecision::Reject {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Reject), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Reject), "REACHABLE_FALSE");
        next.locked = true;
        next.trust = next.trust.saturating_sub(5);
    }
    if decision == VaultDecision::Escalate {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Escalate), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Escalate), "REACHABLE_FALSE");
        next.debt = next.debt.saturating_add(amount / 10);
        next.trust = next.trust.saturating_sub(2);
    }
    if next.debt < 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(next.debt < 0), "REACHABLE_TRUE");
	assert!((next.debt < 0), "REACHABLE_FALSE");
        next.debt = 0;
    }
    if next.liquidity < 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(next.liquidity < 0), "REACHABLE_TRUE");
	assert!((next.liquidity < 0), "REACHABLE_FALSE");
        next.liquidity = 0;
    }
    if next.trust > 100 {
	// KANI INJECTION: Check Reachability
	assert!(!(next.trust > 100), "REACHABLE_TRUE");
	assert!((next.trust > 100), "REACHABLE_FALSE");
        next.trust = 100;
    }
    if next.trust < 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(next.trust < 0), "REACHABLE_TRUE");
	assert!((next.trust < 0), "REACHABLE_FALSE");
        next.trust = 0;
    }

    next
}

pub fn run_vault_rounds(mut snapshot: VaultSnapshot, requests: &[i64], ages: &[u32]) -> i64 {
    let mut idx = 0usize;
    let mut accepted = 0i64;

    while idx < requests.len() {
	// KANI INJECTION: Check Reachability
	assert!(!(idx < requests.len()), "REACHABLE_TRUE");
	assert!((idx < requests.len()), "REACHABLE_FALSE");
        let age = if idx < ages.len() { ages[idx] } else { 30 };
        let decision = evaluate_vault(snapshot, requests[idx], age);

        if decision == VaultDecision::Accept {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Accept), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Accept), "REACHABLE_FALSE");
            accepted = accepted.saturating_add(normalize_amount(requests[idx]));
        }
        if decision == VaultDecision::Reject {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Reject), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Reject), "REACHABLE_FALSE");
            accepted = accepted.saturating_sub(10);
        }
        if decision == VaultDecision::Escalate && snapshot.trust > 50 {
	// KANI INJECTION: Check Reachability
	assert!(!(decision == VaultDecision::Escalate && snapshot.trust > 50), "REACHABLE_TRUE");
	assert!((decision == VaultDecision::Escalate && snapshot.trust > 50), "REACHABLE_FALSE");
            accepted = accepted.saturating_add(5);
        }

        snapshot = apply_decision(snapshot, decision, requests[idx]);
        idx += 1;
    }

    if snapshot.locked {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.locked), "REACHABLE_TRUE");
	assert!((snapshot.locked), "REACHABLE_FALSE");
        accepted = accepted.saturating_sub(25);
    }
    if snapshot.trust > 75 {
	// KANI INJECTION: Check Reachability
	assert!(!(snapshot.trust > 75), "REACHABLE_TRUE");
	assert!((snapshot.trust > 75), "REACHABLE_FALSE");
        accepted = accepted.saturating_add(snapshot.trust as i64);
    }
    if accepted < 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(accepted < 0), "REACHABLE_TRUE");
	assert!((accepted < 0), "REACHABLE_FALSE");
        return 0;
    }

    accepted
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn rejects_locked_vault() {
        let snapshot = VaultSnapshot {
            liquidity: 100,
            debt: 10,
            trust: 90,
            locked: true,
        };
        assert_eq!(evaluate_vault(snapshot, 50, 20), VaultDecision::Escalate);
    }

    #[test]
    fn rounds_do_not_underflow() {
        let snapshot = VaultSnapshot {
            liquidity: 1_000,
            debt: 200,
            trust: 65,
            locked: false,
        };
        let requests = [10, 50, 500, 90, 3];
        let ages = [100, 2, 400];
        assert!(run_vault_rounds(snapshot, &requests, &ages) >= 0);
    }
}
