// Complex Rust contract with conditional logic and loops
// Demonstrates verification of different execution paths

pub struct Account {
    pub balance: i32,
    pub active: bool,
}

impl Account {
    pub fn new(initial_balance: i32) -> Self {
        Account {
            balance: initial_balance,
            active: true,
        }
    }

    pub fn deposit(&mut self, amount: i32) -> bool {
        if amount <= 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(amount <= 0), "REACHABLE_TRUE");
	assert!((amount <= 0), "REACHABLE_FALSE");
            return false;
        }
        if !self.active {
	// KANI INJECTION: Check Reachability
	assert!(!(!self.active), "REACHABLE_TRUE");
	assert!((!self.active), "REACHABLE_FALSE");
            return false;
        }
        self.balance += amount;
        true
    }

    pub fn withdraw(&mut self, amount: i32) -> bool {
        if amount <= 0 {
	// KANI INJECTION: Check Reachability
	assert!(!(amount <= 0), "REACHABLE_TRUE");
	assert!((amount <= 0), "REACHABLE_FALSE");
            return false;
        }
        if !self.active {
	// KANI INJECTION: Check Reachability
	assert!(!(!self.active), "REACHABLE_TRUE");
	assert!((!self.active), "REACHABLE_FALSE");
            return false;
        }
        if self.balance < amount {
	// KANI INJECTION: Check Reachability
	assert!(!(self.balance < amount), "REACHABLE_TRUE");
	assert!((self.balance < amount), "REACHABLE_FALSE");
            return false;
        }
        self.balance -= amount;
        true
    }

    pub fn transfer(&mut self, to: &mut Account, amount: i32) -> bool {
        if !self.withdraw(amount) {
	// KANI INJECTION: Check Reachability
	assert!(!(!self.withdraw(amount)), "REACHABLE_TRUE");
	assert!((!self.withdraw(amount)), "REACHABLE_FALSE");
            return false;
        }
        if !to.deposit(amount) {
	// KANI INJECTION: Check Reachability
	assert!(!(!to.deposit(amount)), "REACHABLE_TRUE");
	assert!((!to.deposit(amount)), "REACHABLE_FALSE");
            // Rollback - this should never happen in normal cases
            self.deposit(amount);
            return false;
        }
        true
    }
}

pub fn process_transaction(account: &mut Account, action: i32, amount: i32) -> i32 {
    match action {
        1 => { // Deposit
            if account.deposit(amount) {
	// KANI INJECTION: Check Reachability
	assert!(!(account.deposit(amount)), "REACHABLE_TRUE");
	assert!((account.deposit(amount)), "REACHABLE_FALSE");
                1 // Success
            } else {
                0 // Failed
            }
        },
        2 => { // Withdraw
            if account.withdraw(amount) {
	// KANI INJECTION: Check Reachability
	assert!(!(account.withdraw(amount)), "REACHABLE_TRUE");
	assert!((account.withdraw(amount)), "REACHABLE_FALSE");
                2 // Success
            } else {
                0 // Failed
            }
        },
        _ => 0, // Invalid action
    }
}

pub fn validate_balance(account: &Account, expected: i32) -> bool {
    account.balance == expected && account.active
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_account_creation() {
        let account = Account::new(100);
        assert_eq!(account.balance, 100);
        assert!(account.active);
    }

    #[test]
    fn test_deposit_valid() {
        let mut account = Account::new(100);
        assert!(account.deposit(50));
        assert_eq!(account.balance, 150);
    }

    #[test]
    fn test_deposit_invalid_amount() {
        let mut account = Account::new(100);
        assert!(!account.deposit(0));
        assert!(!account.deposit(-10));
        assert_eq!(account.balance, 100);
    }

    #[test]
    fn test_withdraw_valid() {
        let mut account = Account::new(100);
        assert!(account.withdraw(30));
        assert_eq!(account.balance, 70);
    }

    #[test]
    fn test_withdraw_insufficient_funds() {
        let mut account = Account::new(50);
        assert!(!account.withdraw(100));
        assert_eq!(account.balance, 50);
    }

    #[test]
    fn test_process_deposit() {
        let mut account = Account::new(100);
        let result = process_transaction(&mut account, 1, 25);
        assert_eq!(result, 1);
        assert_eq!(account.balance, 125);
    }

    #[test]
    fn test_process_withdraw() {
        let mut account = Account::new(100);
        let result = process_transaction(&mut account, 2, 30);
        assert_eq!(result, 2);
        assert_eq!(account.balance, 70);
    }

    #[test]
    fn test_validate_balance() {
        let account = Account::new(200);
        assert!(validate_balance(&account, 200));
        assert!(!validate_balance(&account, 150));
    }
}
